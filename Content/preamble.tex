%%
%% Handle titlepage: Eg the first page which is usually fancy with lots of university logos
%%
\newcommand\ThemeTitlePage{themes/\ThemeName/titlepage.tex}

\IfFileExists{\ThemeTitlePage}{
	\input{\ThemeTitlePage}
	\cleardoublepage
}{}

\IfFileExists{Content/Preamble/colophon.tex}{
	\phantomsection
	\thispagestyle{empty}
	\input{Content/Preamble/colophon.tex}
	\clearpage
}{}

\IfFileExists{Content/Preamble/inscription.tex}{
	\phantomsection
	\thispagestyle{empty}
	\pdfbookmark{Inscription}{Inscription}
	\input{Content/Preamble/inscription.tex}
	\clearpage
}{}

% %%
% %% Include the abstract if any
% %%
\IfFileExists{Content/Preamble/abstract.tex}{
	\phantomsection
	\pdfbookmark{Abstract}{Abstract}
	\begin{abstract}
		\input{Content/Preamble/abstract.tex}
	\end{abstract}
	\cleardoublepage
}{}

\IfFileExists{Content/Preamble/abstract_ita.tex}{
	\pdfbookmark{Sommario}{Sommario}
	\begin{otherlanguage}{italian}
		\begin{abstract}
			\input{Content/Preamble/abstract_ita.tex}
		\end{abstract}
	\end{otherlanguage}

	\cleardoublepage
}{}

\IfFileExists{Content/Preamble/acknowledgements.tex}{
	\phantomsection
	\pdfbookmark{Acknowledgements}{Acknowledgements}

	\begin{vplace}[0.7]
		{
			\centering
			\pgfornament[scale=1.0,color=gray]{113}

			\vspace{1.0cm}

			ACKNOWLEDGEMENTS\par
		}
		\input{Content/Preamble/acknowledgements.tex}
	\end{vplace}

	\cleardoublepage
}{}

% Dedication/inscription
\IfFileExists{Content/Preamble/dedication.tex}{
	\phantomsection
	\thispagestyle{empty}
	\pdfbookmark{Inscription}{Inscription}

	{\mbox{}\vfill \centering \input{Content/Preamble/dedication.tex} \par\vfill}
	\cleardoublepage
}
